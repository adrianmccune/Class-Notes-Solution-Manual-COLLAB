    % Problem
    Ahmad has $22$ carnival ride tickets and it takes $4$ tickets to ride the roller coaster. How many times can he ride the coaster? How many tickets will he have left over?

    \vspace{.3 cm}
    \fbox{\parbox{.6\linewidth}{Number of times he can ride the coaster}} =

    \vspace{.3 cm}
    \fbox{\parbox{.6\linewidth}{Number of tickets left over}} =

    \ifbool{answerKey}{
      \textbf{Answer:}

      Number of times he can ride the coaster = $5$

      Number of tickets left over = $2$

      \textbf{Explanation:}
      \begin{itemize}
          \item Divide the total tickets by tickets per ride: $22 \div 4$.
          \item $4$ goes into $22$ five times (since $4 \times 5 = 20$).
          \item Subtract: $22 - 20 = 2$ tickets remain.
          \item So he can ride 5 times and have 2 tickets left over.
      \end{itemize}
    }{}

    % Problem
    Divide. Give the quotient and remainder.
    $$ 556 \div 8 $$

    \begin{align*}
        \text{Quotient:}& \hspace{1 in}  \\
        \text{Remainder:}& \\
    \end{align*}

    \ifbool{answerKey}{
      \textbf{Answer:}

      Quotient: $69$

      Remainder: $4$

      \textbf{Explanation:}
      \begin{itemize}
          \item Divide 556 by 8.
          \item $8 \times 69 = 552$ (closest multiple without exceeding 556).
          \item Subtract: $556 - 552 = 4$.
          \item So quotient = 69 and remainder = 4.
      \end{itemize}
    }{}

    \vfill\hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%

    \index{({ 1.0\theproblem})\; Rounding to tens or hundreds}
    % Problem
    Round $384$ to the nearest hundred.

    \ifbool{answerKey}{
      \textbf{Answer:}
      $400$

      \textbf{Explanation:}
      \begin{itemize}
          \item Identify the hundreds place: 3.
          \item Look at the digit to the right (tens place): 8.
          \item Since $8 \geq 5$, round up.
          \item So $384$ rounded to the nearest hundred is $400$.
      \end{itemize}
    }{}

    \vfill\hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%
}
