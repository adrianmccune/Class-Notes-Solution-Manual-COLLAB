    \index{({ 1.0\theproblem})\; Introduction to ordering decimals}

    % Problem
    Use $<$, $>$, or $=$ to compare the following decimals.

    \begin{align*}
        1.2 \quad\;\, &\underline{\qquad} \quad 1.07 \\
        \\
        0.36 \quad &\underline{\qquad} \quad 0.54 \\
        \\
        0.50 \quad &\underline{\qquad} \quad 0.5
    \end{align*}

    \ifbool{answerKey}{
      \textbf{Answer:} $>, <, =$

      \textbf{Explanation:}
      \begin{itemize}
          \item Compare 1.2 and 1.07: 1.2 is greater because 0.2 > 0.07.
          \item Compare 0.36 and 0.54: 0.36 is less because 36 hundredths < 54 hundredths.
          \item Compare 0.50 and 0.5: They are equal because both represent the same value.
      \end{itemize}
    }{}

    \vfill\hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%

    % Problem
    Write $0.46$ as a fraction.
    Do not try to simplify your answer.

    \ifbool{answerKey}{
      \textbf{Answer:} $\frac{46}{100}$

      \textbf{Explanation:}
      \begin{itemize}
          \item $0.46$ means 46 hundredths.
          \item Write 46 as the numerator and 100 as the denominator.
          \item So the fraction is $\frac{46}{100}$.
      \end{itemize}
    }{}

    \vfill\hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%

    % Problem
    Write $0.4$ as a fraction in simplest form.

    \ifbool{answerKey}{
      \textbf{Answer:} $\frac{2}{5}$

      \textbf{Explanation:}
      \begin{itemize}
          \item $0.4$ means 4 tenths.
          \item Write as $\frac{4}{10}$.
          \item Simplify by dividing numerator and denominator by 2: $\frac{4}{10} = \frac{2}{5}$.
      \end{itemize}
    }{}

    \vfill\hfill{\scriptsize 1.0\theproblem}

          \item $\frac{62}{100}$ means 62 hundredths.
          \item So the decimal is $0.62$.
      \end{itemize}
    }{}
