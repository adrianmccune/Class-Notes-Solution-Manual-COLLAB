    Find the area of this rectangle.
    \begin{center}
        \includegraphics[width=0.4\linewidth]{images/hippo 86.png}
    \end{center}

    \ifbool{answerKey}{
      \textbf{Answer:} $108 \text{m}^2$

      \textbf{Explanation:}
      \begin{itemize}
          \item The area of a rectangle is found by multiplying its length by its width.
          \item From the figure, the length is 12 meters and the width is 9 meters.
          \item Multiply: $12 \times 9 = 108$.
          \item Include the correct unit: square meters ($\text{m}^2$).
      \end{itemize}
    }{}
