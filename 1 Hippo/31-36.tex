    $$ 4\;, \underline{\qquad}\;, \underline{\qquad}\;, \underline{\qquad} $$

    %type problem above this line

      $$ 4, 6, 8, 10 $$

          \item Start with 4, then add 2: $4 + 2 = 6$.
          \item Add 2 again: $6 + 2 = 8$.
          \item Add 2 once more: $8 + 2 = 10$.
      \end{itemize}
    }{}

    % type problem below this line
    Divide.
    $$ 42 \div 7 $$

    %type problem above this line
    \ifbool{answerKey}{
      \textbf{Answer:}
      $6$

      \textbf{Explanation:}
      \begin{itemize}
          \item Division is the inverse of multiplication.
          \item Find how many times 7 goes into 42.
          \item $7 \times 6 = 42$, so $42 \div 7 = 6$.
      \end{itemize}
    }{}

    \hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%

    % type problem below this line
    Lamar has a bookcase with $40$ books. There are $4$ shelves in the bookcase. Each shelf has the same number of books. How many books does each shelf have?

    %type problem above this line
    \ifbool{answerKey}{
      \textbf{Answer:}
      $10$ books

      \textbf{Explanation:}
      \begin{itemize}
          \item Divide the total number of books by the number of shelves.
          \item $40 \div 4 = 10$.
          \item So each shelf has 10 books.
      \end{itemize}
    }{}

    \hfill{\scriptsize 1.0\theproblem}
    \stepcounter{problem}
  \end{minipage}%

    % type problem below this line

    %type problem above this line
    \ifbool{answerKey}{
      \textbf{Answer:}
      $151$ cans

      \textbf{Explanation:}
      \begin{itemize}
          \item Calculate cans of cola: $18 \times 8 = 144$.
          \item Add Amy's cans: $144 + 7 = 151$.
          \item So they brought 151 cans in all.
      \end{itemize}
    }{}
