    Write $1 \frac{1}{5}$ as an improper fraction.

    \ifbool{answerKey}{
      \textbf{Answer:} $\frac{6}{5}$

      \textbf{Explanation:}
      \begin{itemize}
          \item Multiply the whole number (1) by the denominator (5): $1 \times 5 = 5$.
          \item Add the numerator (1): $5 + 1 = 6$.
          \item Keep the denominator as 5.
          \item So the improper fraction is $\frac{6}{5}$.
      \end{itemize}
    }{}

    % Problem
    Subtract. Write your answer as a mixed number in simplest form.
    $$ 4 \frac{7}{10} - 2 \frac{9}{10} $$

    \ifbool{answerKey}{
      \textbf{Answer:} $1 \frac{4}{5}$

      \textbf{Explanation:}
      \begin{itemize}
          \item Subtract whole numbers: $4 - 2 = 2$.
          \item Subtract fractions: $\frac{7}{10} - \frac{9}{10} = -\frac{2}{10}$.
          \item Borrow 1 from 2: $1 + \frac{10}{10} - \frac{2}{10} = 1 \frac{8}{10}$.
          \item Simplify $\frac{8}{10}$ to $\frac{4}{5}$.
          \item So the answer is $1 \frac{4}{5}$.
      \end{itemize}
    }{}
