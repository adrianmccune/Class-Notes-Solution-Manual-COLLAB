\index{({ 5.0\theproblem} )\; Multiplication with a decimal and a fraction}
% type problem below this line
Evaluate the following.

$$\frac{7}{5} \times 7.95$$

Write your answer as a decimal.





%type problem above this line
\ifbool{answerKey}%
{\begin{center}
Step 1: Convert $\frac{7}{5}$ to a decimal.
$$ \frac{7}{5} means 7 divided by 5.
7 \div 5 = 1.4$$
Step 2: Rewrite.
$$1.4 \times 7.95$$
Step 3: Multiply.
$$1.4 \times 7.95 = 11.13$$
\end {center}}
\vfill \textbf{Answer:}
%type answer below this line
$11.13$



%type answer above this line
}%
