% type problem below this line
Suppose that $Y$ and $Z$ are points on the number line. If $YZ = 17$ and $Y$ lies at $-3$, where could $Z$ be located? \\
If there are several locations, separate them with commas.





%type problem above this line
\ifbool{answerKey}%
{\textbf{Explanation:}
\begin {itemize}
\item Understand that the notation $YZ = 17$ represents the distance between points $Y$ and $Z$ on the number line.
\item Distance on a number line is the absolute value of the difference between the two points.


\vfill \textbf{Answer:}
%type answer below this line
$-20,\; 14$

\index{({ 5.0\theproblem} )\; Additive property of equality with decimals}
% type problem below this line
Solve for $v$.
$$
    v - 3.57 = 4.51
$$





%type problem above this line
\ifbool{answerKey}%
{\textbf{Explanation:}
\begin {itemize}
\item The variable v has 3.57 subtracted from it. To undo that, add 3.57 to both sides.
$$ v - 3.57 + 3.57 = 4.51 + 3.57 $$
\item Simplify.
\end {itemize}}
\vfill \textbf{Answer:}
%type answer below this line
$v = 8.08$

}%

\index{({ 5.0\theproblem} )\; Additive property of equality with integers}
% type problem below this line
Solve for $w$.
$$
    8 + w = 2
$$





%type problem above this line
\ifbool{answerKey}%
{\textbf{Explanation:}
\begin {itemize}
\item The variable w has 8 added to it. To undo that, we subtract 8 from both sides.
$$ 8 + w - 8 = 2 - 8$$
\item Simplify.
\end {itemize}}
\vfill \textbf{Answer:}
%type answer below this line
$w = -6 $
