% type problem below this line
A water dispenser has a capacity of 25,000 milliliters. What is its capacity in liters?
Be sure to include the correct unit in your answer.





%type problem above this line
\ifbool{answerKey}%
{\textbf{Explanation:}
\begin {itemize}
\item Identify units for conversion, milliliters (mL) to liters (L).
\item Identify conversion factor (1 mL = how many liters).
$$1000 mL = 1 L$$ so,
$$1 mL = \frac {1}{1000} L $$
\item Multiply by number of mL, which is 25000.
$$25000 mL = \frac {1}{1000} L \times 25000$$
\end {itemize}}
\vfill \textbf{Answer:}
%type answer below this line
$25$ L

width = $0.2 \times 1000 m = 200 m$
\item Plug in converted values to area formula.
Area = length \times width = $350 m \times 200 m$
\item Multiply.
\end {itemize}}
\vfill \textbf{Answer:}

\index{({ 5.0\theproblem} )\; Multiplicative property of equality with fractions}
% type problem below this line
Solve for $x$. 
$$
14 = \frac{7}{9}x
$$

Simplify your answer as much as possible.
 


%type problem above this line
\ifbool{answerKey}%
{\textbf{Explanation:}
\begin {itemize}
\item The variable $x$ is divided by $9$, so we undo that by multiplying both sides by $9$.
$$14 \times 9 = \frac{7}{9}x \times 9
126 = 7 x$$
\item Now $x$ is being multiplied by 7, so we undo that by dividing both sides by 7.
$$\frac {126}{7} = \frac {7}{7}x $$
\end {itemize}}
\vfill \textbf{Answer:}
%type answer below this line
$x = 18$
